let formFields = [];

function addField(){

    const label =
        document.getElementById("fieldLabel").value;

    const type =
        document.getElementById("fieldType").value;

    const required =
        document.getElementById("requiredField").checked;

    if(!label){
        alert("Enter field label");
        return;
    }

    formFields.push({
        label,
        type,
        required
    });

    renderForm();

    document.getElementById("fieldLabel").value="";
}

function renderForm(){

    const form =
        document.getElementById("dynamicForm");

    form.innerHTML="";

    formFields.forEach(field=>{

        const label =
            document.createElement("label");

        label.innerText =
            field.label;

        form.appendChild(label);

        let input;

        switch(field.type){

            case "textarea":
                input =
                    document.createElement("textarea");
                break;

            case "checkbox":
                input =
                    document.createElement("input");
                input.type="checkbox";
                break;

            default:
                input =
                    document.createElement("input");
                input.type=field.type;
        }

        if(field.required){
            input.required=true;
        }

        input.name=field.label;
        input.id = field.label;

        form.appendChild(input);

        form.appendChild(
            document.createElement("br")
        );
    });
}

function uploadFile(){

    const file =
        document.getElementById("fileInput").files[0];

    if(!file){
        alert("Select file");
        return;
    }

    const allowed =
    [
        "application/pdf",
        "image/png",
        "image/jpg",
        "image/jpeg"
    ];

    if(!allowed.includes(file.type)){

        document.getElementById(
            "uploadStatus"
        ).innerText =
        "Unsupported file";

        return;
    }

    document.getElementById(
        "uploadStatus"
    ).innerText =
    "File uploaded successfully";
}

function autofillForm(data){

    formFields.forEach(field=>{

        const element =
        document.getElementById(
            field.label
        );

        if(!element) return;

        const value =
        data[field.label];

        if(value){

            if(
                element.type ===
                "checkbox"
            ){
                element.checked=true;
            }
            else{
                element.value=value;
            }

        }
        else{

            element.style.border =
            "2px solid red";

        }

    });

}
